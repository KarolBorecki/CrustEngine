196void Projector::RecalculateProjectionMatrix(Camera &cam)
197{
198 static double q;
199 q = cam.GetFFar() / (cam.GetFFar() - cam.GetFNear());
200 r_projMat[0][0] = _aspectRatio * cam.GetFFovRad();
201 r_projMat[1][1] = cam.GetFFovRad();
202 r_projMat[2][2] = q;
203 r_projMat[3][2] = 1.0;
204 r_projMat[2][3] = -cam.GetFNear() * q;

313Matrix<float>& Projector::CalculateTranslationMatrix(Transform& transform)
314{
315 static Matrix<float> translationMat(4, 4, 0.0); // TODO transform matrxi does not need to be calculated for each poli!!!
316 translationMat.MakeIdentity();
317 translationMat[0][3] = transform.GetPosition().X();
318 translationMat[1][3] = transform.GetPosition().Y();
319 translationMat[2][3] = transform.GetPosition().Z();
320 translationMat[3][3] = 1.0f;
321
322 static Matrix<float> scaleMat(4, 4, 0.0); // TODO transform matrxi does not need to be calculated for each poli!!!
323 scaleMat[0][0] = transform.GetScale().X();
324 scaleMat[1][1] = transform.GetScale().Y();
325 scaleMat[2][2] = transform.GetScale().Z();
326 scaleMat[3][3] = 1.0f;
327
328 static Matrix<float> rotationZMat(4, 4, 0.0);
329 rotationZMat[0][0] = Math::Cos(transform.GetEulerRotation().Z());
330 rotationZMat[0][1] = Math::Sin(transform.GetEulerRotation().Z());
331 rotationZMat[1][0] = -Math::Sin(transform.GetEulerRotation().Z());
332 rotationZMat[1][1] = Math::Cos(transform.GetEulerRotation().Z());
333 rotationZMat[2][2] = 1.0f;
334 rotationZMat[3][3] = 1.0f;
335
336 static Matrix<float> rotationXMat(4, 4, 0.0);
337 rotationXMat[0][0] = 1.0f;
338 rotationXMat[1][1] = Math::Cos(transform.GetEulerRotation().X());
339 rotationXMat[1][2] = Math::Sin(transform.GetEulerRotation().X());
340 rotationXMat[2][1] = -Math::Sin(transform.GetEulerRotation().X());
341 rotationXMat[2][2] = Math::Cos(transform.GetEulerRotation().X());
342 rotationXMat[3][3] = 1.0f;
343
344 static Matrix<float> rotationYMat(4, 4, 0.0); // Not tested
345 rotationYMat[0][0] = Math::Cos(transform.GetEulerRotation().Y());
346 rotationYMat[0][2] = Math::Sin(transform.GetEulerRotation().Y());
347 rotationYMat[2][0] = -Math::Sin(transform.GetEulerRotation().Y());
348 rotationYMat[1][1] = 1.0f;
349 rotationYMat[2][2] = Math::Cos(transform.GetEulerRotation().Y());
350 rotationYMat[3][3] = 1.0f;
351
352 static Matrix<float> worldMat(4, 4, 0.0);
353 worldMat.MakeIdentity();
354 worldMat = rotationZMat;
355 worldMat *= rotationXMat;
356 worldMat *= rotationYMat;
357 worldMat *= translationMat;
358 worldMat *= scaleMat;
359
360 r_transMat = worldMat;
361 return r_transMat;
