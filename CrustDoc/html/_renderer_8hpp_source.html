240void Renderer::RecalculateProjectionMatrix(Camera* cam) {
241 double q = cam->GetFFar() / (cam->GetFFar() - cam->GetFNear());
242 projMatrix->PutValue(0, 0, aspectRatio * cam->GetFFovRad());
243 projMatrix->PutValue(1, 1, cam->GetFFovRad());
244 projMatrix->PutValue(2, 2, q);
245 projMatrix->PutValue(2, 3, 1.0);
246 projMatrix->PutValue(3, 2, -cam->GetFNear() * q);
247 Logger::Info(projMatrix->ToString().c_str());
