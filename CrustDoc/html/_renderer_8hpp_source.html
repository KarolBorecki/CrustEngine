251void Renderer::RecalculateProjectionMatrix(Camera &cam)
252{
253 double q = cam.GetFFar() / (cam.GetFFar() - cam.GetFNear());
254 (*projMatrix)[0][0] = aspectRatio * cam.GetFFovRad();
255 (*projMatrix)[1][1] = cam.GetFFovRad();
256 (*projMatrix)[2][2] = q;
257 (*projMatrix)[3][2] = 1.0;
258 (*projMatrix)[2][3] = -cam.GetFNear() * q;
259}
